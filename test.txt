"class Solution {\n    public int[] twoSum(int[] nums, int target) {\n        Map<Integer, Integer> diff = new HashMap<>();\n        for(int i=0; i<nums.length; i++){\n            int d = target - nums[i];\n            if(diff.containsKey(nums[i])){\n                // return null;\n                return new int[]{diff.get(nums[i]), i};\n            }\n            diff.put(d, i);\n        }\n        return null;\n    }\n}"
