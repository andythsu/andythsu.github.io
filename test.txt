class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> diff = new HashMap<>();
        for(int i=0; i<nums.length; i++){
            int d = target - nums[i];
            if(diff.containsKey(nums[i])){
                // return null;
                return new int[]{diff.get(nums[i]), i};
            }
            diff.put(d, i);
        }
        return null;
    }
}
